int frac(int n) {
    if (n == 1) {
        return 1;
    } else {
        return n * frac( n - 1 );
    }
    return 0;
}

char fu(int s, int b) {
    int x;
    x = getint();
    if (s > b) { x = -x; } else {x = x % 3;}
    return x;
}

int main() {
    printf("%d%d\n", frac(10), fu(9, 'a'));
    return 0;
}
