int main() {
    int x = 0;
    int y = 5;

    if (y > 3) {
        x = 10;
    } else {
        x = 20;
    }

int i;
    for (i = 0; i < 5; i = i + 1) {
        if (i % 2 == 0) {
            x = x + i;
        } else {
            x = x * 2;
        }
    }

    printf("Final value of x: %d\n", x);
    return 0;
}
